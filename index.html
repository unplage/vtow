async function loadWhisperPipeline() { if (pipeline) return pipeline; if (isModelLoading) { showToast('模型正在加载中，请稍后...'); return null; } isModelLoading = true; modelStatus.innerText = '⏳ 正在下载/加载模型（首次需下载约150MB，请耐心等待）...'; progressContainer.style.display = 'block'; progressFill.style.width = '0%'; try { if (!self.transformers || !self.transformers.pipeline) { throw new Error('Transformers.js 未正确加载，请刷新页面重试。'); } const { pipeline: transformersPipeline } = self.transformers; pipeline = await transformersPipeline('automatic-speech-recognition', 'Xenova/whisper-tiny', { progress_callback: (progress) => { if (progress.status === 'downloading' && progress.total) { const percent = (progress.loaded / progress.total) * 100; progressFill.style.width = percent + '%'; modelStatus.innerText = `⏳ 下载中 ${Math.round(percent)}%`; } else if (progress.status === 'done') { progressFill.style.width = '100%'; modelStatus.innerText = '✅ 模型已就绪，可上传音频'; setTimeout(() => { progressContainer.style.display = 'none'; }, 1000); } } }); showToast('模型加载完成，现在可以离线识别'); } catch (e) { console.error('模型加载失败:', e); modelStatus.innerText = '❌ 模型加载失败：' + e.message; showToast('模型加载失败: ' + e.message); progressContainer.style.display = 'none'; pipeline = null; } finally { isModelLoading = false; } return pipeline; }
